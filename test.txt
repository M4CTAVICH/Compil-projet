    INTEGER somme , i;
}

    somme = 0;

    }

    WRITELN(somme);

    i = 1;
    WHILE (i <= 5) {

        i = i + 1;
        FOR (i:1:1:i<=15) {
        somme = somme + i;
        
    }
    }

END
